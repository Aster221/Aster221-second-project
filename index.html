import pandas as pd # 读取销售数据 data = pd.read_csv('Sales.csv') # 1. 每种产品的总销售额 total_sales_per_product = data.groupby('product')['sales'].sum() print("每种产品的总销售额：") print(total_sales_per_product) # 2. 最畅销的产品 best_selling_product = total_sales_per_product.idxmax() best_selling_product_sales = total_sales_per_product.max() print(f"\n最畅销的产品是: {best_selling_product}，总销售额为: {best_selling_product_sales}") # 3. 销售额最高的一天 sales_per_day = data.groupby('date')['sales'].sum() highest_sales_day = sales_per_day.idxmax() highest_sales_value = sales_per_day.max() print(f"\n销售额最高的一天是: {highest_sales_day}，销售额为: {highest_sales_value}")
